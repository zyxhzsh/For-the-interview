#### 描述

中缀表达式支持加减乘除和括号，求表达式的值。

#### 方法

（1）将中缀表达式转换成后缀表达式[](https://github.com/GrowTowardsSunlight/For-the-interview/blob/master/algorithm/中缀表达式转后缀表达式.md)

（2）后缀表达式求值

```java
import java.util.*;


public class Solution {
    
    //中缀表达式求值
    public int solve (String s) {
        
        ArrayList<String> strs = infixToSuffix(s);//中缀表达式转后缀表达式
        
        //后缀表达式求值
        Stack<Integer> stack = new Stack<>();
        for(int i=0;i<strs.size();i++){
            String cur = strs.get(i);
            if(!cur.equals("+") && !cur.equals("-") && !cur.equals("*") && !cur.equals("/")){
                stack.push(Integer.parseInt(cur));
            }else{
                int b = stack.pop();
                int a = stack.pop();
                switch(cur){
                    case "+":
                        stack.push(a+b);
                        break;
                    case "-":
                        stack.push(a-b);
                        break;
                    case "*":
                        stack.push(a*b);
                        break;
                    default:
                        stack.push(a/b);
                }
            }
        }
        return stack.peek();
    }
    
    private ArrayList<String> infixToSuffix(String s){
        
        ArrayList<String> list = new ArrayList<>();
        Stack<Character> stack = new Stack<>();
        char[] chars = s.toCharArray();
        int n = chars.length;
        
        for(int i=0;i<n;i++){
            char c = chars[i];
            if(Character.isDigit(c)){
                StringBuilder sb = new StringBuilder();
                sb.append(c);
                while(i < n - 1 && Character.isDigit(chars[i+1])){
                    sb.append(chars[i+1]);
                    i++;
                }
                list.add(sb.toString());
            }else if(c == '('){
                stack.push('(');
            }else if(c == ')'){
                while(stack.peek() != '('){
                    list.add(""+stack.pop());
                }
                stack.pop();
            }else{
                while(!stack.isEmpty() && stack.peek() != '(' && priority(stack.peek()) >= priority(c)){
                    list.add(""+stack.pop());
                }
                stack.push(c);
            }
        }
        while(!stack.isEmpty()){
            list.add(""+stack.pop());
        }
        return list;
    }
    
    private int priority(char c){
        if(c == '*' || c == '/'){
            return 1;
        }else{
            return -1;
        }
    }
}
```
